Example 1

# Dr Seuss
a <- 1:2
fish.number <- paste(a, "fish", sep ="_")
b <- c("red", "blue")
fish.color <- paste(b, "fish", sep = "_")
print(c(fish.number, fish.color))
## [1] "1_fish"    "2_fish"    "red_fish"  "blue_fish"
